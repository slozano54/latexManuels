\begin{exercice*}[Quatrième proportionnelle $\mathbb{D}$]
    Déterminer la quatrième proportionnelle dans les égalités suivantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \begin{spacing}{2}
                \item $\dfrac{0{,}8}{2}=\dfrac{t}{0{,}4}$
                \item $\dfrac{0{,}4}{z}=\dfrac{2}{0{,}7}$
            \end{spacing}
        \end{enumerate}
    \end{multicols}

\end{exercice*}

    \begin{multicols}{2}
    \begin{enumerate}
        \begin{spacing}{1.5}
            \item 
            $\dfrac{0{,}8}{2}=\dfrac{t}{0{,}4}$\\
            {\bfseries \color[HTML]{f15929}Les produits en croix sont égaux.}\\
            $2\times t = 0{,}8\times 0{,}4$\\
            {\bfseries \color[HTML]{f15929}On divise les deux membres par 2}.\\
            $\dfrac{2\times t}{2}= \dfrac{0{,}8\times 0{,}4}{2}$\\
            {\bfseries \color[HTML]{f15929}On simplifie et on calcule.}\\
            $t=0{,}16$            
            \item 
            $\dfrac{0{,}4}{z}=\dfrac{2}{0{,}7}$\\
            {\bfseries \color[HTML]{f15929}Les produits en croix sont égaux.}\\
            $2\times z = 0{,}7\times 0{,}4$\\
            {\bfseries \color[HTML]{f15929}On divise les deux membres par 2}.\\
            $\dfrac{2\times z}{2}= \dfrac{0{,}7\times 0{,}4}{2}$\\
            {\bfseries \color[HTML]{f15929}On simplifie et on calcule.}\\
            $z=0{,}14$     
        \end{spacing}
    \end{enumerate}
    \end{multicols}
\end{corrige}
