\begin{exercice*}[Quatrième proportionnelle $\mathbb{N}$]
    Déterminer la quatrième proportionnelle dans les égalités suivantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \begin{spacing}{2}
                \item $\dfrac{8}{2}=\dfrac{y}{4}$
                \item $\dfrac{9}{t}=\dfrac{5}{4}$
            \end{spacing}
        \end{enumerate}
    \end{multicols}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}
    Déterminer la quatrième proportionnelle dans les égalités suivantes.
    \begin{multicols}{2}
    \begin{enumerate}
        \begin{spacing}{1.2}
            \item 
        $\dfrac{8}{2}=\dfrac{y}{4}$\\
        {\bfseries \color[HTML]{f15929}Les produits en croix sont égaux.}\\
        $2\times y = 8\times 4$\\
        {\bfseries \color[HTML]{f15929}On divise les deux membres par 2}.\\
        $\dfrac{2\times y}{2}= \dfrac{8\times 4}{2}$\\
        {\bfseries \color[HTML]{f15929}On simplifie et on calcule.}\\
        $y=16$        
            \item 
        $\dfrac{9}{t}=\dfrac{5}{4}$\\
        {\bfseries \color[HTML]{f15929}Les produits en croix sont égaux.}\\
        $5\times t = 4\times 9$\\
        {\bfseries \color[HTML]{f15929}On divise les deux membres par 5}.\\
        $\dfrac{5\times t}{5}= \dfrac{4\times 9}{5}$\\
        {\bfseries \color[HTML]{f15929}On simplifie et on calcule.}\\
        $t=7{,}2$        
        \end{spacing}
    \end{enumerate}
    \end{multicols}
\end{corrige}
