\begin{exercice*}[\ExerciceRefMethode{N2methode4}]
    \label{N2exosBase012}
    Décomposer chaque nombre en produit de deux facteurs premiers.
    \begin{enumerate}
        \item $51$
        \item $55$
        \item $106$
        \item $141$
    \end{enumerate}

    \hrefMathalea{https://coopmaths.fr/mathalea.html?ex=4A11-1,s=false,s2=2,n=3,i=0&v=l}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}
    Décompositions en produit de deux facteurs premiers.
    
    \begin{enumerate}
        \item $51=3\times 17$
        \item $55=5\times 11$
        \item $106=2\times 53$
        \item $141=3\times 47$
    \end{enumerate}    
\end{corrige}
