\begin{exercice*}[\ExerciceRefMethode{N2methode4}]
    \label{N2exosBase014}
    Décomposer chaque nombre en produit de facteurs premiers.
    \begin{enumerate}
        \item $292$
        \item $425$
        \item $873$
        \item $\num{1900}$
    \end{enumerate}

    \hrefMathalea{https://coopmaths.fr/mathalea.html?ex=4A11-1,s=false,s2=4,n=3,i=0&v=l}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}    
    Décompositions en produit de facteurs premiers.
    \begin{enumerate}
        \item $292=2^2\times 73$
        \item $425=5^2\times 17$
        \item $873=3^2\times 97$
        \item $\num{1900}=2^2\times 5^2 \times 19$
    \end{enumerate}

\end{corrige}
