\begin{exercice*}[\ExerciceRefMethode{N2methode4}]
    \label{N2exosBase014}
    Décomposer chaque nombre en produit de facteurs premiers.
    \begin{itemize}
        \item[] $A=292$
        \item[] $B=425$
        \item[] $C=873$
        \item[] $D=\num{1900}$
    \end{itemize}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}    
    Décompositions en produit de facteurs premiers.
    
    \begin{itemize}
        \def\item{}
        \item $A=292=2^2\times 73$
        \item $B=425=5^2\times 17$
        \item $C=873=3^2\times 97$
        \item $D=\num{1900}=2^2\times 5^2 \times 19$
    \end{itemize}

\end{corrige}
