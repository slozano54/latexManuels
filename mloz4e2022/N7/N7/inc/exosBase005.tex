\begin{exercice*}
    Déterminer si le nombre \num{4} est solution de chacune de ces équations.
    \begin{enumerate}
        \item $5x-6=3x+2$
        \item $x^2-9=3x-5$
        \item $\dfrac{x-1}{12}=\dfrac{1}{4}$
    \end{enumerate}
\end{exercice*}
