\begin{exercice*}
    Relier chaque équation de gauche à sa solution de droite.
    \begin{center}
        \Relie[LargeurG=3cm,Ecart=1cm]{%
        $2x+3=3x+7$/$-4$,%
        $2x+3=-3x+7$/\num{-0.8},%
        $2x-3=-3x+7$/$0$,%
        $2x-3=-3x-7$/\num{0.8},%
        $2x-3=3x-7$/$2$,%
        $2x=3x$/$4$
        }
    \end{center}
\end{exercice*}

        }

\end{corrige}
