\begin{exercice*}
    Relier chaque équation de gauche à sa solution de droite.
    \begin{center}
        \Relie[LargeurG=2cm,Ecart=1cm]{%
        $3x+1=-2$/$1$,%
        $3x-1=-2$/$\dfrac{2}{3}$,%
        $3x=2$/$\dfrac{1}{3}$,%
        $3x-1=2$/$-\dfrac{1}{3}$,%
        $3x+1=2$/$-1$%
        }
    \end{center}
\end{exercice*}
