\begin{exercice*}
    Soit le programme de calcul suivant.

    \myProgCalcul{$\leadsto$}{Programme}{%
        \ProgCalcul[Enonce,ThemePerso]{%
            Choisis un nombre,%
            Soustrais 8 à ce nombre,%
            Multiplie le résultat par -4,%
            Ajoute le quadruple du nombre de départ,%
        }
    }

    \begin{enumerate}
        \item Exécute ce programme de calcul :
              \begin{itemize}
                  \item pour $x = 3$
                  \item pour $x = -2$
              \end{itemize}

        \item Que remarques-tu ?

        \item Quelle expression obtiens-tu si le nombre de départ est $x$ ?

        \item Explique alors ta réponse à la question précédente.
    \end{enumerate}
\end{exercice*}
