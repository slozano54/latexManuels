\begin{exercice*}
    Voici deux programmes de calculs.

        \myProgCalcul{$\leadsto$}{Programme 1}{%
        \ProgCalcul[Enonce,ThemePerso]{%
            Choisir un nombre,%
            Calculer le carré de ce nombre,%
            Soustraire le double du nombre choisi au départ,%
            }
        }
        \myProgCalcul{$\leadsto$}{Programme 2}{%
        \ProgCalcul[Enonce,ThemePerso]{%
            Choisir un nombre,%
            Soustraire 2,%
            Multiplier le résultat obtenu par le nombre choisi au départ,%
            }
        }

    \begin{enumerate}
        \item Teste le programme 1 pour $x = 0$ et $x = 1$.
        \item Teste le programme 2 pour $x = 0$ et $x = 1$.
        \item Que constates-tu ?
        \item Soit $x$ le nombre choisi au départ. Exprime le résultat des deux programmes en fonction de $x$ et montre que les programmes sont équivalents.
    \end{enumerate}
\end{exercice*}
