    \[6x^{2} + 4x = 2x \times 3x + 2x \times 2\]

        \begin{enumerate}
            \item $13 \times 4,5 + 4,5 \times x$
            \item $5x - 4x + 3x$
            \item $7a + a^{2} - 6a$
            \item $9y^{2} - 6y + 3y$
            \item $12x^{2} + 6x + 18$
            \item $-2n^{2} - 4n - 6$
            \item $1,7y^{2} - 3,4y$
        \end{enumerate}
    \end{multicols}
\end{exercice*}
