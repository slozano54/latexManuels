\begin{exercice*}
    \begin{enumerate}
        \item On considère l'expression $A = 3x + 5x(x - 2)$.

              \begin{itemize}
                  \item Souligne l'opération prioritaire.
                  \item Développe et réduis l'expression A.
              \end{itemize}

        \item On considère l'expression $B = 4 - 2(3 - 5u)$.

              \begin{itemize}
                  \item Complète : $B = 4 + \cdots + (3 - 5u)$.
                  \item Développe et réduis l'expression B.
              \end{itemize}

        \item On considère l'expression $C = 3x - (2x + 5) \times 4$.

              \begin{itemize}
                  \item Mets des crochets autour de l'opération prioritaire.
                  \item Développe et réduis l'expression C.
              \end{itemize}
    \end{enumerate}
\end{exercice*}
