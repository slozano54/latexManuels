\begin{exercice*}
    Écrire chaque nombre sous la forme d'une fraction puis d'un nombre décimal si c'est possible.
    \begin{multicols}{2}
        \begin{spacing}{1.8}
            \begin{list}{}{}
                \item $A=\dfrac{1}{\dfrac{1}{5}}$
                \item $B=\dfrac{1}{\dfrac{1}{\num{1.35}}}$
                \item $C=\dfrac{1}{\dfrac{1}{19}}$
                \item $D=\dfrac{1}{\dfrac{1}{-8}}$
                \item $E=\dfrac{1}{\dfrac{7}{4}}$
                \item $F=\dfrac{1}{\dfrac{-19}{20}}$
                \item $G=\dfrac{1}{\dfrac{\num{6.2}}{\num{3.4}}}$
                \item $H=\dfrac{1}{-\dfrac{7}{12}}$
            \end{list}               
        \end{spacing}
    \end{multicols}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}    
    Écrire chaque nombre sous la forme d'une fraction puis d'un nombre décimal si c'est possible.
    \begin{multicols}{2}
        \begin{spacing}{1.5}
            \begin{itemize}
                \def\item{}
                \item $A=\dfrac{1}{\dfrac{1}{5}}$\\\smallskip
                \item {\red $A=5$}\\
                \item $B=\dfrac{1}{\dfrac{1}{\num{1.35}}}$\\\smallskip
                \item {\red $B=\num{1.35}$}\\
                \item $C=\dfrac{1}{\dfrac{1}{19}}$\\\smallskip
                \item {\red $C=19$}\\
                \item $D=\dfrac{1}{\dfrac{1}{-8}}$\\\smallskip
                \item {\red $D=-8$}\\
                \item $E=\dfrac{1}{\dfrac{7}{4}}$\\\smallskip
                \item {\red $E=\dfrac{4}{7}$}\\
                \item $F=\dfrac{1}{\dfrac{-19}{20}}$\\\smallskip
                \item {\red $F=\dfrac{-20}{19}$}\\                                
                \item $G=\dfrac{1}{\dfrac{\num{6.2}}{\num{3.4}}}$\\\smallskip
                \item {\red $G=\dfrac{\num{3.4}}{\num{6.2}}=\dfrac{34}{62}=\dfrac{17}{31}$}\\\smallskip
                \item $H=\dfrac{1}{-\dfrac{7}{12}}$\\\smallskip
                \item {\red $H=\dfrac{-12}{7}$}
            \end{itemize}               
        \end{spacing}
    \end{multicols}
\end{corrige}
