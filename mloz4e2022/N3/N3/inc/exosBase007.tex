\begin{exercice*}    
    \begin{enumerate}
        \item Décomposer les couples suivants en produit de facteurs premiers :
        \begin{multicols}{2}                
            \begin{enumerate}
                \item $34$ et $86$.
                \item $145$ et $265$
                \item $105$ et $63$
                \item $207$ et $135$
            \end{enumerate}
        \end{multicols}
        \item Simplifier alors les fractions suivantes :
        \begin{multicols}{2}
            \begin{spacing}{2}
                \begin{enumerate}
                    \item $\dfrac{34}{86}$.
                    \item $\dfrac{265}{145}$.
                    \item $\dfrac{63}{105}$.
                    \item $\dfrac{207}{135}$.
                \end{enumerate}                    
            \end{spacing}
        \end{multicols}
    \end{enumerate}                   
    
\end{exercice*}
