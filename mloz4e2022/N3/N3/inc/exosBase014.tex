\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    % \phantom{rrr}    
    Mettre au même dénominateur puis comparer.

    \begin{enumerate}
        \begin{spacing}{2}
            \item $\dfrac{-11}{16}$ et $\dfrac{-17}{24}$.
            
            {\red $\dfrac{-11}{16}=\dfrac{-11\times3}{16\times3}=-\dfrac{33}{48}$ et $\dfrac{-17}{24}=\dfrac{-17\times2}{24\times2}=-\dfrac{34}{48}$.
            
            $-\dfrac{33}{48}>-\dfrac{34}{48}$ donc $\dfrac{-11}{16}>\dfrac{-17}{24}$.
            }
            \item $\dfrac{\num{8.25}}{27}$ et $\dfrac{\num{-5.5}}{\num{-18}}$.
            
            {\red En mettant sur $54$, on obtient que $\dfrac{\num{8.25}}{27}=\dfrac{\num{-5.5}}{\num{-18}}$.}
            \item $\dfrac{-17}{71}$ et $\dfrac{71}{-17}$.
            
            {\red En comparant avec $-1$, on obtient que $\dfrac{-17}{71}<-1<\dfrac{71}{-17}$ donc $\dfrac{-17}{71}<\dfrac{71}{-17}$.}
            \item $-\dfrac{11}{8}$ et $\dfrac{-9}{5}$.            
            
            {\red En mettant sur $40$, on obtient que $-\dfrac{11}{8}>\dfrac{-9}{5}$.}
        \end{spacing}
    \end{enumerate}

\end{corrige}
