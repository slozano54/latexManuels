        \begin{itemize}
            \def\item{}
            \begin{spacing}{2}
                \item $A=-\dfrac{9}{5}+\dfrac{7}{5}$\\
                {\red $A=\dfrac{-9+7}{5}$\\
                $A=-\dfrac{2}{5}$\\
                }
                \item $B=\dfrac{12}{25}-\dfrac{-17}{25}+\dfrac{-133}{25}$\\
                {\red $B=\dfrac{12+17-133}{25}$\\
                $B=-\dfrac{104}{25}$\\
                }                
                \item $C=-5+\dfrac{6}{-5}$\\
                {\red $C=\dfrac{-25}{5}+\dfrac{-6}{5}$\\
                $C=\dfrac{-31}{5}$\\
                }
                \item $D=-\dfrac{5}{21}-\dfrac{7}{3}$\\
                {\red $D=-\dfrac{5}{21}-\dfrac{7}{3}$\\
                $D=\dfrac{-5-49}{21}$\\
                $D=\dfrac{-54}{21}=\dfrac{-18}{7}$\\
                }
            \end{spacing}
        \end{itemize}
        \Coupe
