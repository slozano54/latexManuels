    \begin{multicols}{3}
        \begin{spacing}{1.5}
            \begin{enumerate}
                \item $\dfrac{1}{3}   {\red \neq} \num{0.33}$
                \item $\dfrac{11}{4}  {\red =} \num{2.75}$
                \item $\dfrac{13}{5}  {\red =} \num{2.6}$
                \item $\dfrac{57}{11} {\red \neq} \num{5.181}$
                \item $\dfrac{43}{12} {\red \neq} \num{3.583}$
                \item $\dfrac{26}{16} {\red =} \num{1.625}$
            \end{enumerate}
        \end{spacing}
    \end{multicols}

\end{corrige}
