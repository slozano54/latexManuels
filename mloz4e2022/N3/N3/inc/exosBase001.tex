\begin{exercice*}
    Donner l'écriture décimale de chaque fraction.
    \begin{multicols}{3}        
        \begin{enumerate}
            \begin{spacing}{1.5}
                \item $\dfrac{1}{2}$
                \item $\dfrac{3}{2}$
                \item $\dfrac{1}{4}$
                \item $\dfrac{15}{4}$
                \item $\dfrac{4}{5}$
                \item $\dfrac{14}{5}$              
            \end{spacing}
        \end{enumerate}         
    \end{multicols}
\end{exercice*}

                \item $\dfrac{14}{5}{\red = \num{2.8}}$
            \end{spacing}
        \end{enumerate}        
    \end{multicols}
\end{corrige}
