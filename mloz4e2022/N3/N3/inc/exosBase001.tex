\begin{exercice*}
    Donner l'écriture décimale de chaque fraction.
    \begin{multicols}{3}
        \begin{spacing}{1.5}
            \begin{enumerate}
                \item $\dfrac{1}{2}$
                \item $\dfrac{3}{2}$
                \item $\dfrac{1}{4}$
                \item $\dfrac{13}{4}$
                \item $\dfrac{4}{5}$
                \item $\dfrac{12}{5}$
            \end{enumerate}           
        \end{spacing}
    \end{multicols}
\end{exercice*}

                \item $\dfrac{13}{4}{\red = \num{3.25}}$
                \item $\dfrac{4}{5} {\red = \num{0.8}}$
                \item $\dfrac{12}{5}{\red = \num{2.4}}$
            \end{enumerate}           
        \end{spacing}
    \end{multicols}
\end{corrige}
