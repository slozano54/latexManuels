\begin{exercice*}
    \begin{enumerate}
        \item Déterminer le nombre positif dont le carré est $121$.
        \item Calculer de tête $\sqrt{49}$.
        \item Calculer de tête $\sqrt{100}$.
        \item Déterminer le nombre positif dont le carré est $9$.
    \end{enumerate}
