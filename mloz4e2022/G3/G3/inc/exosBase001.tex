\begin{exercice*}
    Calculer les carrés des nombres suivants :
    \begin{multicols}{3}
        \begin{enumerate}
            \begin{spacing}{1.5}
                \item $1$
                \item $6$
                \item $\num{0.3}$
                \item $-2$
                \item $\dfrac{5}{3}$
                \item $-\dfrac{4}{7}$
            \end{spacing}
        \end{enumerate}
    \end{multicols}

    \hrefMathalea{https://coopmaths.fr/mathalea.html?ex=4G20-3,s=4,n=4&v=l}
\end{exercice*}

    Calculer les carrés des nombres suivants :
    \begin{multicols}{2}
        \begin{enumerate}
            \begin{spacing}{1.5}
                \item ${\red 1^2 = 1}$
                \item ${\red 6^2 = 36}$
                \item ${\red \num{0.3}^2 = \num{0.09}}$
                \item ${\red (-2)^2 = 4}$
                \item ${\red \left(\dfrac{5}{3}\right)^2 = \dfrac{25}{9}}$
                \item ${\red \left(-\dfrac{4}{7}\right)^2 = \dfrac{16}{49}}$
            \end{spacing}
        \end{enumerate}
    \end{multicols}
\end{corrige}
