\begin{exercice*}[Signe d'un produit de deux relatifs]
    Donner le signe des produits suivants.
    \begin{multicols}2
        \begin{enumerate}
            \item $ (-13) \times (-10) $
            \item $ (+20) \times (-11) $
            \item $ (+19) \times (+11) $
            \item $ (-1) \times (+12) $
            \item $ (+14) \times (-2) $
            \item $ (-8) \times (-1) $
            \item $ (+17) \times (-4) $
            \item $ (-4) \times (-10) $
        \end{enumerate}
    \end{multicols}

    \hrefAleaTeX{https://urls.mathslozano.fr/4n12025ex14}
\end{exercice*}
