\begin{exercice*}[Déclinaison de produit]
    Calculer sachant que $11,2\times 2,5 = 28$.
    \begin{multicols}2
        \begin{enumerate}
            \item $ 11,2\times (-2,5) $
            \item $ -11,2\times (-2,5) $
            \item $ 1,12\times (-25) $
            \item $ 112\times (-25) $
        \end{enumerate}
    \end{multicols}    
\end{exercice*}

    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés

            \item $ 11,2\times (-2,5) = -28 $
            \item $ -11,2\times (-2,5) = 28 $
            \item $ 1,12\times (-25) = -28 $
            \item $ 112\times (-25) = \num{2800}$
        \end{enumerate}
    \end{multicols}
\end{corrige}
