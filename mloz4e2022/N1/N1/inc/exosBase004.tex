\begin{exercice*}[Additionner des entiers relatifs]
    Effectuer les additions suivantes.
    \begin{multicols}2
        \begin{enumerate}            
                \item $ (+19) + (-20)$
                \item $ (+20) + (-2)$
                \item $ (-8) + (+5)$
                \item $ (-3) + (+18)$
                \item $ (-15) + (-10)$
                \item $ (+2) + (-3)$
                \item $ (-14) + (+15)$
                \item $ (-11) + (-19)$            
        \end{enumerate}
    \end{multicols}
\end{exercice*}
