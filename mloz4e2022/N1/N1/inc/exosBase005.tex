\begin{exercice*}[Additionner des décimaux relatifs]
    Effectuer les additions suivantes.
    \begin{multicols}2
        \begin{enumerate}            
                \item $ (+3,5) + (-1,5)$
                \item $ (-5,4) + (-10,4)$
                \item $ (+1,9) + (-8,3)$
                \item $ (+8,35) + (+17,2)$
                \item $ (-0,84) + (+1)$
                \item $ (-3,25) + (-5,75)$
        \end{enumerate}
    \end{multicols}    
\end{exercice*}
