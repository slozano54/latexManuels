\begin{exercice*}[Comparer des décimaux]
    Comparer les nombres suivants.
    \begin{multicols}2
        \begin{enumerate}
            \item $+3,5$ et $-4,5$
            \item $-4,1$ et $-4,7$
            \item $-8,8$ et $-8,42$
            \item $-20,3$ et $+20,2$
            \item $+4,25$ et $+4,2$
            \item $-2,25$ et $-2,205$
            \item $-2,12$ et $-4,7$
            \item $0,01$ et $-0,01$
        \end{enumerate}
    \end{multicols}
\end{exercice*}
