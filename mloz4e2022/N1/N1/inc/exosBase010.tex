\begin{exercice*}[Sommes de deux décimaux relatifs simplifiées]
    Calculer.
    \begin{multicols}2
        \begin{enumerate}
            \item $ -2,5-4,2 $
            \item $ -1,7+4,2 $
            \item $ 0,2-0,9 $
            \item $ -3,25-3,75 $
            \item $ -6+10,6 $
            \item $ 3,8-1,1 $
            \item $ -1,62+0,62 $
            \item $ -5,5-50,5 $
        \end{enumerate}
    \end{multicols}    
\end{exercice*}

\end{corrige}
