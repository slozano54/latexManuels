\begin{exercice*}
   Un robinet laisse échapper de façon continue trois litres d’eau en deux heures.
   \begin{enumerate}
      \item Quelle quantité d’eau s'écoulera en une demi-journée ?
      \item Quel temps s’est écoulé pour laisser s’échapper \Capa[L]{51} ?
   \end{enumerate}
\end{exercice*}
\begin{corrige}
   \begin{enumerate}
      \item En une demi-journée, il s'écoule \Capa[L]{18} d'eau.
      \item Il s'est écoulé 34 heures pour laisser s'échapper \Capa[L]{51}.
   \end{enumerate}
\end{corrige}
