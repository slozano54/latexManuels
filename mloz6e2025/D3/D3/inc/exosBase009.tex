\begin{exercice*}
   Trois professeurs de mathématiques ont corrigé cent copies en deux heures.
   \begin{enumerate}
      \item Combien de professeurs pour 50 copies en 20 minutes ?
      \item Combien de temps avec 9 professeurs pour ces 100 copies ?
   \end{enumerate}
\end{exercice*}
\begin{corrige}
   \begin{enumerate}
      \item Il faut un professeur pour 50 copies en 20 minutes.
      \item Avec 9 professeurs, il faut environ 13 minutes pour corriger ces 100 copies.
   \end{enumerate}
\end{corrige}
