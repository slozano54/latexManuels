\begin{corrige}
   \begin{enumerate}
      \item Non, car $\dfrac{10}{15} \neq \dfrac{30}{50}$.
      \item Oui, car $\dfrac{20}{50} = \dfrac{60}{150} = \dfrac{80}{200}$.
      \item Non, car $\dfrac{123,35}{765,87} \neq \dfrac{1\,354,76}{1\,236,23}$.
      \item Oui, car $\dfrac{9}{9,9} = \dfrac{10}{11} = \dfrac{13}{14,3}$.
   \end{enumerate}
\end{corrige}
