\begin{exercice*}
   Pour télécharger un fichier de 40 Mo (mégaoctets), un ordinateur met \Temps{;;;;;80}.
   \begin{enumerate}
      \item Combien de temps pour télécharger un fichier de 1 Mo ?
      \item Quelle est la taille d'un fichier téléchargé en une seconde ?
   \end{enumerate}
\end{exercice*}
\begin{corrige}
   \begin{enumerate}
      \item Pour télécharger un fichier de 1 Mo, l'ordinateur met \Temps{;;;;;2}.
      \item La taille d'un fichier téléchargé en une seconde est de \Masse{0,5}.
   \end{enumerate}
\end{corrige}
