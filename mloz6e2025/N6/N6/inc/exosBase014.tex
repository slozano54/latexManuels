         \hline
      \end{tabular}
      \qquad 
      \begin{tabular}{|>{\centering\arraybackslash}p{0.6cm}|>{\centering\arraybackslash}p{0.6cm}|>{\centering\arraybackslash}p{0.6cm}|}

         & \num{0.2} & \num{0.125} \\

         \hline
      \end{tabular}}
   \end{center}
