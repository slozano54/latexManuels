\begin{prerequis}[Connaisances \emoji{red-heart} et compétences \emoji{diamond-suit} du cycle 3]    
    \begin{itemize}
        \item[\emoji{red-heart}] Connaître et utiliser les termes : expérience aléatoire, issue, événement, certain, impossible, probable, improbable.
        \item[\emoji{red-heart}] Savoir qu’une probabilité s’exprime comme un nombre entre 0 et 1 (ou en pourcentage).
        \item[\emoji{red-heart}] Comprendre qu’une probabilité mesure la chance qu’un événement a de se produire.
        \item[\emoji{diamond-suit}] Lister toutes les issues possibles d’une expérience aléatoire simple (ex. : lancer de dé, pile ou face).
        \item[\emoji{diamond-suit}] Comparer des probabilités d’événements simples (ex. : « il y a plus de chances d’obtenir un nombre pair qu’un 6 avec un dé »).
    \end{itemize}
\end{prerequis}
