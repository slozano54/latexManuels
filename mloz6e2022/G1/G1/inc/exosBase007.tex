\begin{exercice} %7
    Vrai (V) ou faux (F) ?
    \begin{enumerate}
       \item Si $C \in (AB)$ alors $A \in (BC)$.
       \item Si $E \in [DF]$ alors $D \in [EF]$.
       \item Si $C \in [AB)$ mais $C \notin [AB]$ alors $A \in [CB)$.
       \item Si $C \in [BA)$ mais $C \notin [AB]$ alors $B \in [AC)$.
    \end{enumerate}
 \end{exercice}
