\begin{exercice}
    Recopier et compléter les égalités suivantes.
    \begin{multicols}{2}
        \begin{spacing}{1.5}
            \begin{enumerate}
                \item $\num{4.5}+\dots =5$
                \item $\num{7.8}+\dots=8$
                \item $\num{0.8}+\dots=1$
                \item $\dots+\num{0.23}=1$
                \item $\dots+\num{5.8}=6$
                \item $\dots-\num{2.3} =2$
                \item $\dots-\num{0.9} =4$
                \item $\dots-\num{5.8} =7$
                \item $\num{7.3}-\dots =7$
                \item $8-\dots =\num{7.6}$
            \end{enumerate}
        \end{spacing}
    \end{multicols}
 \end{exercice}
