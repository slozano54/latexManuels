   \begin{multicols}{3}
      \begin{enumerate}
         \item $\dfrac{27}2$ \bigskip
         \item $\dfrac{92}8$ \bigskip
         \item $\dfrac{13}{3}$ \bigskip
         \item $\dfrac{53}{3}$ \bigskip
         \item $\dfrac{119}{4}$ \bigskip
         \item $\dfrac{127}{4}$
      \end{enumerate}
   \end{multicols}
\end{exercice*}
