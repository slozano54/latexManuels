\section{Périmètre des polygones}
\begin{propriete}
    Le \textbf{périmètre} d'un polygone est égal à la somme des longueurs de ses côtés.
\end{propriete}

\begin{remarque}
    Il est possible d'établir des formules pour certains polygones particuliers.
\end{remarque}
