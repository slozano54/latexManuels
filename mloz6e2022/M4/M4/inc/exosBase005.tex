      \begin{pspicture}(0,-0.7)(5,2.5)

         \rput(-0.3,-0.3){D}
         \rput(-0.3,2.3){A}
         \rput(5.3,0){C}
         \rput(2,2.3){B}
         \rput(1.5,-0.3){M}
         \rput(1,2.3){2 cm}
         \rput{90}(-0.3,1){\Lg[cm]{3}}

      \item Déterminer la position du point $M$ pour que ces deux périmètres soient égaux.
   \end{enumerate}
