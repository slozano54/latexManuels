\begin{exercice}
   Calculer le volume :
   \begin{enumerate}
      \item d'un pavé droit possédant deux faces opposées carrées de côté \Lg[cm]{5} et une hauteur de \Lg[cm]{7}.
      \item d'un cube de côté \Lg[dm]{2.5}.
      \item d'un pavé droit dont la hauteur est de \Lg[cm]{9}, la largeur mesure la moitié de la hauteur et la longueur est le triple de la hauteur.
   \end{enumerate}
\end{exercice}
