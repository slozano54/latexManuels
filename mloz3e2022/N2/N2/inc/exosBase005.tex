\begin{exercice*}
Citer tous les nombres compris entre $20$ et $50$, à la fois divisibles par :  
    \begin{enumerate}
        \item $3$ et $5$.
        \item $2$ et $9$.
        \item $3$ et $6$.
    \end{enumerate}
\end{exercice*}
