\begin{exercice*}
    Dans cet exercice, toutes les rotations sont de centre $A$.\\
    \begin{tikzpicture}[scale=1]
        \coordinate (A) at (4,4);
        \tkzLabelPoint[below,shift={(0,-0.2)}](A){$A$};
        \coordinate[label=above left:$C$] (C) at (4,6);
        \coordinate[label=above:$B$] (B) at (4,7);        
        \tkzDrawCircle(A,B);
        \tkzDrawCircle(A,C);
        %%%%
        \tkzDefPointsBy[rotation=center A angle 36](B,C){T,U};        
        \tkzLabelPoint[above left](T){$T$};
        \tkzLabelPoint[above](U){$U$};
        %%%%
        \tkzDefPointsBy[rotation=center A angle 36](T,U){R,S};
        \tkzLabelPoint[left](R){$R$};
        \tkzLabelPoint[above](S){$S$};   
        %%%%
        \tkzDefPointsBy[rotation=center A angle 36](R,S){P,Q};
        \tkzLabelPoint[left](P){$P$};
        \tkzLabelPoint[above left](Q){$Q$};
        %%%%
        \tkzDefPointsBy[rotation=center A angle 36](P,Q){N,O};
        \tkzLabelPoint[below left](N){$N$};
        \tkzLabelPoint[below left](O){$O$};
        %%%
        \tkzDefPointsBy[symmetry=center A](B,C,T,U,R,S,P,Q,N,O){L,M,J,K,H,I,F,G,D,E};
        \tkzLabelPoint[below](L){$L$};
        \tkzLabelPoint[below left](M){$M$};
        \tkzLabelPoint[below right](J){$J$};
        \tkzLabelPoint[below right](K){$K$};
        \tkzLabelPoint[right](H){$H$};
        \tkzLabelPoint[right](I){$I$};
        \tkzLabelPoint[right](G){$G$};
        \tkzLabelPoint[right](F){$F$};
        \tkzLabelPoint[above right](D){$D$};
        \tkzLabelPoint[above right](E){$E$};
        \tkzDrawSegments(B,L T,J R,H P,F N,D);
    \end{tikzpicture}
\end{exercice*}
