    \begin{enumerate}

            \coordinate[label=below right:$B$] (B) at (5,3);
            \tkzDrawPoints[shape=cross out,size=3pt](A,B);
        \end{tikzpicture}
        \item Construire $F$, l'image de $E$ par la rotation de centre $D$ et d'angle \ang{90}.\\

            \coordinate[label=above right:$D$] (D) at (5,4);
