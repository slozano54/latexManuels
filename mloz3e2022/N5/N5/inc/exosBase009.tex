    \begin{enumerate}
        \item Factoriser $C$.
        \item Résoudre $C=0$.
    \end{enumerate}
\end{exercice*}
\begin{corrige}
    Soit $C=(3x+2)^2 - 49$
    
    \begin{enumerate}
        \item Factoriser $C$.
        
        $C=C=(3x+2)^2 - 7^2$\\
        $C=[(3x+2)+7]\times[(3x+2)-7]$\\
        $C=(3x+2+7)\times(3x+2-7)$\\
        $C=(3x+9)(3x-5)$
        \item Résoudre $C=0$.
        
        \ResolEquation[Produit,Facteurs,Entier,Simplification]{3}{9}{3}{-5}
    \end{enumerate}
\end{corrige}
