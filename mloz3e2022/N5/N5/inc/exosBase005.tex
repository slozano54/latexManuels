\begin{exercice*}
    Rédoudre chaque équation
    \begin{enumerate}
        \item $(-3x+2)(x+9)=0$
        \item $(6x+5)(4x-9)=0$
        \item $(7x-3)(-5x-11)=0$
    \end{enumerate}
\end{exercice*}
\begin{corrige}
    Rédoudre chaque équation

    \begin{enumerate}
        \item $(-3x+2)(x+9)=0$
        
        \ResolEquation[Produit,Facteurs,Entier,Simplification]{-3}{2}{1}{9}
        \item $(6x+5)(4x-9)=0$
        
        \ResolEquation[Produit,Facteurs,Entier,Simplification]{6}{5}{4}{-9}
    \end{enumerate}
    \Coupe
    \begin{enumerate}
        \setcounter{enumi}{2}
        \item $(7x-3)(-5x-11)=0$
        
        \ResolEquation[Produit,Facteurs,Entier,Simplification]{7}{-3}{-5}{-11}
    \end{enumerate}
\end{corrige}
