\begin{exercice*}
    Résoudre chaque équation. Donner les solutions exactes sous la forme $\sqrt{a}$ et $-\sqrt{a}$.
    \begin{multicols}{2}
        \begin{enumerate}
            \item $x^2=7$
            \item $y^2=\num{9.1}$
            \item $z^2=30$
            \item $x^2-3=0$
            \item $y^2-\num{2.7}=0$
            \item $z^2-\num{0.5}=0$
        \end{enumerate}
    \end{multicols}
\end{exercice*}
