\begin{exercice}
    Convertir \Temps{;;;;;6000} en heures-minutes-secondes.
\end{exercice}
\begin{corrige}
    \Temps{;;;;;6000} = $1\times $ \Temps{;;;;;3600} + $40\times $ \Temps{;;;;;60}
    % $\num{6000} = 1\times\num{3600} + 40\times 60$
    
    \Temps{;;;;;6000} = \Temps{;;;1;;} et \Temps{;;;;;60} = \Temps{;;;;1;}
    % $\num{3600}$ s $=1$ h et $\num{60}$ s $=1$ minutes
    
    donc \Temps{;;;;;6000} $=$ \Temps{;;;1;40;} 
    % donc $\num{6000}$ secondes = $1$ heure et $40$ minutes.
\end{corrige}
