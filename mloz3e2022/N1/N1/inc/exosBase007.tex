\begin{exercice*}[Partage]
    On partage une somme de \Prix[0]{17500} entre $3$ personnes. La première reçoit les $\dfrac25$
    de la somme totale, la deuxième reçoit les $\dfrac34$ de la part de la première.
    \begin{enumerate}
        \item Quelle fraction de la somme totale revient à la troisième personne ?
        \item Calcule la part de chacun.
    \end{enumerate}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    \phantom{rrr}    
    \begin{multicols}2
        \begin{enumerate}
            \begin{spacing}2
            \item $F = 1-\dfrac{2}{5}-\dfrac{3}{4}\times \dfrac{2}{5}$
            
            $F = \dfrac{10}{10}-\dfrac{4}{10}-\dfrac{3}{10}$

            $F = \dfrac{3}{10}$
            \item Part du $1$\up{er} : $\dfrac{2}{5}\times \num{17500}=\dfrac{\num{35000}}{5}= \num{7000}$
            
            Part du $2$\up{ème} : $\dfrac{3}{4}\times \num{7000}=\dfrac{\num{21000}}{4}= \num{5250}$

            Part du $3$\up{ème} : $\dfrac{3}{10}\times \num{17500}=3\times \num{1750}= \num{5250}$

            Vérification : $\num{7000}+\num{5250}+\num{5250}=\num{17500}$
            \end{spacing}
        \end{enumerate}
    \end{multicols}
\end{corrige}
