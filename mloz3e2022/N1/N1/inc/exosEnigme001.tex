% Les enigmes ne sont pas numérotées par défaut donc il faut ajouter manuellement la numérotation
% si on veut mettre plusieurs enigmes
\refstepcounter{exercice}
\numeroteEnigme
\begin{enigme}
    Les nombres en bout de ligne ou de colonne sont les sommes des nombres contenus dans la ligne ou la colonne.

    Compléter les grilles avec des nombres qui conviennent (plusieurs solutions possibles).

    \Yohaku[Perso]{%
        $\dfrac{1}{9}$/$\dfrac{5}{9}$/$\dfrac{2}{3}$/$\dfrac{5}{9}$,
        $\dfrac{7}{9}$/$\dfrac{10}{9}$/$\dfrac{11}{9}$/$\dfrac{2}{3}$
    }    
    \hfill
    \Yohaku[Perso]{%
        $3$/$\dfrac{4}{3}$/$\dfrac{2}{3}$/$\dfrac{1}{3}$,
        $\dfrac{11}{3}$/$\dfrac{5}{3}$/$\dfrac{13}{3}$/$1$
    }    
    \hfill
    \Yohaku[Perso]{%
        $\dfrac{9}{2}$/$\dfrac{3}{2}$/$4$/$\dfrac{5}{2}$,
        $\dfrac{17}{2}$/$4$/$\dfrac{13}{2}$/$6$
    }    
\end{enigme}
