\begin{exercice}
    $(-3)+(+12)=$~\ldots
    \begin{ChoixQCM}{4}
        \item $(+9)$
        \item $(+15)$
        \item $(-15)$
        \item $(-9)$
    \end{ChoixQCM}
\end{exercice}
\begin{corrige}
    Réponse \reponseQCM{a}. En effet :
    \begin{itemize}
        \item $(-3)$ et $(+12)$ sont de signes contraires et $12>3$ donc la somme sera du même signe que $(+12)$, soit positive.
        \item La différence des distances à zéro vaut $12-3=9$
    \end{itemize}
\end{corrige}
