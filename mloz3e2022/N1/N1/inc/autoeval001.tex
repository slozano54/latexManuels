\begin{exercice}
    Calculer.
    \begin{enumerate}
        \item $(15-10)\div5$
        \item $15-10\div5$
        \item $(10-15)\times 5$
        \item $10-15\times 5$
    \end{enumerate}
\end{exercice}
\begin{corrige}
    \begin{enumerate}
        \item $(15-10)\div5 = 5\div 5 = 1$
        \item $15-10\div5 = 15-2 = 13$
        \item $(10-15)\times 5 =-5\times 5 = -25$
        \item $10-15\times 5 = 10 -3 = 7$
    \end{enumerate}
\end{corrige}
