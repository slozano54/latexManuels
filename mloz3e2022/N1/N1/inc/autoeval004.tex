\begin{exercice}
    Un avion décolle à \Temps{;;;8;06;} et atterrit \\ à \Temps{;;;16;54;}.
    
    Quelle a été la durée du vol ?
\end{exercice}
\begin{corrige}
    \Temps{;;;16;54;} $-$ \Temps{;;;8;06;} = \Temps{;;;8;48;}
    % $16\text{h}54\text{min}-8\text{h}06\text{min} = 8\text{h}48\text{min}$.
\end{corrige}
