\begin{exercice*}
    Calculer rapidement.
    \begin{multicols}{2}
        \begin{itemize}
            \item[] $A=101^2-99^2$
            \item[] $B=235^2-234^2$
            \item[] $C=105^2-95^2$
            \item[] $D=\num{9875}^2-125^2$
        \end{itemize}
    \end{multicols}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    %\phantom{rrr}    
    Calculer rapidement.
    
        \begin{itemize}
            \def\item{}
            \item $A=101^2-99^2$\\
            {\red 
            $A=(101+99)\times (101-99)$\\
            $A=200\times 2$\\
            $A=400$
            }\\

            \smallskip
            \item $B=235^2-234^2$\\
            {\red 
            $B=(235+234)\times (235-234)$\\
            $B=469\times 1$\\
            $B=469$
            }\\

            \smallskip
        \end{itemize}
        \Coupe
        \begin{itemize}
            \def\item{}
            \item $C=105^2-95^2$\\
            {\red 
            $C=(105+95)\times (105-95)$\\
            $C=200\times 10$\\
            $C=\num{2000}$
            }\\

            \smallskip
            \item $D=\num{9875}^2-125^2$\\
            {\red 
            $D=(\num{9875}+125)\times (\num{9875}-125)$\\
            $D=\num{10000}\times \num{9750}$\\
            $D=\num{97 500 000}$
            }\\

            \smallskip
        \end{itemize}
        % \vspace*{-20mm}
\end{corrige}
