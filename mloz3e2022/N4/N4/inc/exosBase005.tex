\begin{exercice*}
    Supprimer les parenthèses et réduire les expressions suivantes.
    \begin{multicols}{2}
        \begin{itemize}
            \item[] $A=-(3c-2)$
            \item[] $B=(-3a+4)$
            \item[] $C=(6c^2-6c+4)$
            \item[] $D=-(-5c^2-9c+3)$
            \item[] $E=-(7x+9)$
            \item[] $F=-(3a^2-4a+11)$
        \end{itemize}
    \end{multicols}
    \hrefMathalea{https://coopmaths.fr/mathalea.html?ex=3L10-2,s=1-2-3-4,n=6,i=0&v=l}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    %\phantom{rrr}    
    Supprimer les parenthèses et réduire les expressions suivantes.
    \begin{multicols}{2}
        \begin{itemize}
            \def\item{}
            \item $A=-(3c-2)$\\
            {\red $A=-3c+2$\\\smallskip}
            \item $B=(-3a+4)$\\
            {\red $B=-3a+4$\\\smallskip}
            \item $C=(6c^2-6c+4)$\\
            {\red $C=6c^2-6c+4$\\\smallskip}
            \item $D=-(-5c^2-9c+3)$\\
            {\red $D=5c^2+9c-3$\\\smallskip}
            \item $E=-(7x+9)$\\
            {\red $E=-7x-9$\\\smallskip}
            \item $F=-(3a^2-4a+11)$\\
            {\red $F=-3a^2+4a-11$}
        \end{itemize}
    \end{multicols}
\end{corrige}
