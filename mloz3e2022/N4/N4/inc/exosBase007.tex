\begin{exercice*}
    Supprimer les parenthèses et réduire les expressions suivantes.
    \begin{itemize}
        \begin{spacing}{1.2}
            \item[] $A= (-5z+4) - (5z-3)$
            \item[] $B= -(-6c-1) + (4c+5)$
            \item[] $C= (9c-6) - (5c+8)$
            \item[] $D= -(8c-1) + (-6c-9)$
        \end{spacing}
    \end{itemize}    
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    %\phantom{rrr}    
    Supprimer les parenthèses et réduire les expressions suivantes.
    
    \begin{itemize}
        \def\item{}
        \item $A= (-5z+4) - (5z-3)$\\
        {\red 
        $A= -5z+4  -5z+3$\\
    $A=-10z+7$\\\smallskip
        }
        \item $B= -(-6c-1) + (4c+5)$\\
        {\red 
        $B= 6c+1 + 4c+5$\\
        $B=10c+6$\\\smallskip
        }
        \item $C= (9c-6) - (5c+8)$\\
        {\red 
        $C= 9c-6  -5c-8$\\
        $C=4c-14$\\\smallskip
        }
        \item $D= -(8c-1) + (-6c-9)$\\
        {\red 
        $D= -8c+1  -6c-9$\\
        $D=-14c-8$
        }
    \end{itemize}
\end{corrige}
