\begin{exercice*}
    Déterminer les expressions égales.
    \begin{multicols}{2}
        \begin{itemize}
            \begin{spacing}{1.1}
                \item[] $A= 8x+3-(6x+2)$
                \item[] $B= (9x+5)+(-2x+3)$
                \item[] $C= (4x-9)-(2x-7)$
                \item[] $D= (9x+5)-2x+3$
                \item[] $E= (4x-9)-2x+7$
                \item[] $F= 8x+3-6x-2$
            \end{spacing}
        \end{itemize}
    \end{multicols}
\end{exercice*}
\begin{corrige}
    %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
    %\phantom{rrr}    
    Déterminer les expressions égales.
    \begin{multicols}{2}
        \begin{itemize}
            \begin{spacing}{1.2}
                \def\item{}
                \item $A= 8x+3-(6x+2)$\\
                {\red $A = 8x+3-6x-2$\\$A=2x+1$}\\
                \item $B= (9x+5)+(-2x+3)$\\
                {\red $B= 9x+5-2x+3$\\$B=7x+8$}\\
                \item $C= (4x-9)-(2x-7)$\\
                {\red $C=4x-9-2x+7$\\$C=2x-2$}\\
                \columnbreak
                \item $D= (9x+5)-2x+3$\\
                {\red $D=7x+8$}\\
                \item $E= (4x-9)-2x+7$\\
                {\red $E=4x-9-2x+7$\\$E=2x-2$}\\
                \item $F= 8x+3-6x-2$\\
                {\red $F=2x+1$}
            \end{spacing}
        \end{itemize}
    \end{multicols}
    {\red Donc $A=F$ ; $B=D$ et $C=E$.}
\end{corrige}
