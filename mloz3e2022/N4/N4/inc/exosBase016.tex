    \begin{enumerate}
        \item $A=(2x-3)(4+x)$\\
        
            \begin{tabular}{|*{3}{@{}>{\vrule width0pt height\dimexpr.65cm-.2pt\relax depth\dimexpr.35cm-.2pt\relax\centering\arraybackslash}p{\dimexpr2cm-.4pt\relax}@{}|}}            
               \hline               
               $\times$ &$2x$           &$-3$           \\\hline
               $4$      &{\red $8x$}    &{\red $-12$}   \\\hline
               $+x$     &{\red $2x^2$}  &{\red $-3x$}   \\\hline            
            \end{tabular}

        \bigskip 
        Développement : {\red $A=8x+2x^2-12-3x$}

        \medskip
        Développement réduit : {\red $A=2x^2+5x-12$}
