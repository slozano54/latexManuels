\begin{exercice}
    Convertir les durées données ci-dessous en heures et minutes.
    \begin{enumerate}
       \item $\Temps{;;;1,5;;}$.
       \item $\Temps{;;;2,25;;}$.
       \item $\Temps{;;;0,3;;}$.
    \end{enumerate}
    \hrefMathalea{https://coopmaths.fr/mathalea.html?ex=6D101,n=5,i=0&v=l}
 \end{exercice}
 
 \begin{corrige}
    \ \\ [-5mm]
    \begin{enumerate}
       \item $\Temps{;;;1,5;;} {\red = \Temps{;;;1;30;}.}$
       \item $\Temps{;;;2,25;;}{\red = \Temps{;;;2;15;}.}$
       \item $\Temps{;;;0,3;;} {\red = \num{0.3}\times 60 = \Temps{;;;;18;}.}$
    \end{enumerate}
 \end{corrige}
