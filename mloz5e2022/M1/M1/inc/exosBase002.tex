\begin{exercice}
   Convertir les durées données ci-dessous en heures décimales.
   \begin{enumerate}
      \item $\Temps{;;;6;30;}$.
      \item $\Temps{;;;2;45;}$.
      \item $\Temps{;;;8;33;}$.
   \end{enumerate}
\end{exercice}

\begin{corrige}
   \ \\ [-5mm]
   \begin{enumerate}
      \item $\Temps{;;;6;30;}{\red = \Temps{;;;6,5;;}}$.
      \item $\Temps{;;;2;45;}{\red = \Temps{;;;2,75;;}}$.
      \item $\Temps{;;;8;33;}{\red = 8 + \dfrac{33}{60} = \Temps{;;;8,55;;}}$.
   \end{enumerate}
\end{corrige}
