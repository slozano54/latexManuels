\begin{exercice*}
    Colorier avec une couleur différente chaque paire \textbf{d'angles alternes-internes}.
    \par\smallskip
    \begin{center}
        \begin{Geometrie}[CoinHD={(3.5u,3u)}]
            pair A[],B[];
            A0=u*(1.75,2);
            B0=u*(2.25,1);
            A1=A0 shifted (u,-0.2u);
            A2=A0 shifted (-u,0.25u);
            A3=1.3[B0,A0];
            B1=B0 shifted (u,0.2u);
            B2=B0 shifted (-u,-0.25u);
            B3=1.3[A0,B0];
            trace droite(A0,B0);
            trace droite(A2,A1);
            trace droite(B2,B1);
        \end{Geometrie}
    \end{center}
\end{exercice*}
