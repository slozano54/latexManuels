\begin{exercice*} %9
   Trois poules pondent dix \oe ufs en deux heures.
   \begin{enumerate}
      \item Combien de poules faudrait-il pour pondre cinq \oe ufs en vingt minutes ?
      \item Combien de temps mettraient neuf poules pour pondre vingt \oe ufs ?
   \end{enumerate}
\end{exercice*}

\begin{corrige}
   3 poules pondent 10 \oe ufs en \Temps{;;;2} donc : \\
   \begin{enumerate}
      \item 3 poules pondent 5 \oe ufs en $\Temps{;;;1} =\Temps{;;;;60}$ ; \\
         donc, {\red 9 poules pondent 5 \oe ufs en \Temps{;;;;20}} (il faut 3 fois plus de poules pour 3 fois moins de temps). \\
      \item 9 poules pondent 5 \oe ufs en \Temps{;;;;20} ; \\
         donc, {\red 9 poules pondent 20 \oe ufs en \Temps{;;;;80}} (il y a 4 fois plus d'\oe ufs, il faut 4 fois plus de temps). \\ [5cm]
   \end{enumerate}
\end{corrige}
