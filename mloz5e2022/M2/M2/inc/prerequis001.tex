    \begin{itemize}  
        \item Calculer périmètre et aire des figures usuelles.
        \item Calculer périmètre et aire d’un assemblage de figures.

        \item Effectuer des conversions d’unités de longueurs et d’aires.
    \end{itemize}
\end{prerequis}
