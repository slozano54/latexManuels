\begin{exercice*}
   Le périmètre d'un triangle non aplati est de \Lg[cm]{18}. Ce triangle peut-il avoir un côté\dots
   \begin{multicols}{2}
      \begin{enumerate}
         \item de \Lg[cm]{7} ?
         \item de \Lg[cm]{4} ?
         \item de \Lg[cm]{11} ?
         \item de \Lg[cm]{9} ?
      \end{enumerate}
   \end{multicols}
   Justifier en donnant un exemple lorsque cela est possible ou en le prouvant dans le cas contraire.

   \hrefMathalea{https://coopmaths.fr/alea/?uuid=f789c&id=5G21-1&n=3&d=10&s=2&alea=WrcL&cd=1&v=eleve&es=02110}
\end{exercice*}
\begin{corrige}
   %\setcounter{partie}{0} % Pour s'assurer que le compteur de \partie est à zéro dans les corrigés
   \begin{enumerate}
      \item Avec un côté de \Lg[cm]{7}, il reste \Lg[cm]{11}. On peut choisir les deux autres côtés de mesures \Lg[cm]{6} et \Lg[cm]{5} par exemple. {\red Ce triangle est  constructible} puisque $\Lg[cm]{6}+\Lg[cm]{5} =\Lg[cm]{11} >\Lg[cm]{7}$. \smallskip
      \item Avec un côté de \Lg[cm]{4}, il reste \Lg[cm]{14}. On peut choisir les deux autres côtés de mesures \Lg[cm]{7} et \Lg[cm]{7} par exemple. {\red Ce triangle est constructible} puisque $\Lg[cm]{7}+\Lg[cm]{4} =\Lg[cm]{11} >\Lg[cm]{7}$. \smallskip
      \item Avec un côté de \Lg[cm]{11}, il reste \Lg[cm]{7}. On ne peut pas construire un triangle car \Lg[cm]{7} est inférieur à \Lg[cm]{11}. {\red Ce triangle n'est pas constructible}.
   \end{enumerate}
   \Coupe
   \begin{enumerate}      
      \item Avec un côté de \Lg[cm]{9}, il reste \Lg[cm]{9}. On peut choisir les deux autres côtés de mesures \Lg[cm]{4} et \Lg[cm]{5} par exemple. {\red Ce triangle est un triangle plat} puisque $\Lg[cm]{4}+\Lg[cm]{5} =\Lg[cm]{9}$.
   \end{enumerate}
\end{corrige}
