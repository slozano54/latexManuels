    \begin{multicols}{2}
      \begin{enumerate}
         \item $12-5\times3$
         \item $12\times(5+3)$
         \item $(12-5)\div3$
         \item $12+\dfrac{5}{3}$
      \end{enumerate}
   \end{multicols}
 \end{exercice}
