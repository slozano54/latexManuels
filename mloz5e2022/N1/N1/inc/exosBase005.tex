\begin{exercice}%5
    Traduire les expressions suivantes en français.
    \begin{multicols}{2}
      \begin{enumerate}
         \item $12-5\times3$
         \item $12\times(5+3)$
         \item $(12-5)\div3$
         \item $12+\dfrac{5}{3}$
      \end{enumerate}
   \end{multicols}
 \end{exercice}

       \item \textcolor{red}{Le quotient de la différence de 12 et de 5 par }3
       \item \textcolor{red}{La somme de 12 et du quotient de 5 par 3}.
    \end{enumerate}
