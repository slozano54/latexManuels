\begin{exercice*}%6
    Traduire l'enchaînement d'opérations des programmes suivants à l'aide d'une expression puis les calculer.
    \setlength{\columnseprule}{0pt}
    \begin{multicols}{2}
       \ProgCalcul[Enonce,Largeur=3cm]{Prendre 7,Ajouter 2, Multiplier par 3, Soustraire 3}
       \ProgCalcul[Enonce,Largeur=3cm]{Prendre 6,Multiplier par 7, Diviser par 3,Soustraire 4}
    \end{multicols}
 \end{exercice*}
 
 \begin{corrige}
    Premier programme : \textcolor{red}{$(7 + 2)\times3-4$} \\ [1mm]
    \ProgCalcul[Ecart=11mm]{7,+2 *3 -3}. On trouve \textcolor{red}{24}. \\
    Deuxième programme :  \textcolor{red}{$6\times7\div3-4$} \\ [1mm]
    \ProgCalcul[Ecart=11mm]{6,*7 /3 -4}. On trouve \textcolor{red}{10}.
 \end{corrige}
