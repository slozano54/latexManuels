\begin{exercice*}
   Effectuer les calculs suivants en simplifiant.
   \begin{itemize}
      \item[] $A =(+12) + (-11) + (+25) + (-17)$
      \item[] $B = (-2,1) + (-9) + (+6,4) + (-8,3)$
      \item[] $C = (+14) + (-7) + (+2) + (-3,75) + (-5,25)$
      \item[] $D = (+13,5) + (-8,1) + (-6,9) + (-5,5)$
      \item[] $E=(-7)+(+1)+(-10)$
   \end{itemize}
   \hrefMathalea{https://coopmaths.fr/alea/?uuid=f6ea7&id=5R22&n=5&d=10&s=20&s2=false&cd=1&v=confeleve&v=eleve&title=&es=02110}
\end{exercice*}

\begin{corrige}
   \ \\ [-5mm]
   \begin{itemize}
      \def\item{}
      \item $A =(+12) + (-11) + (+25) + (-17)$ \\
         \qquad\; $=+(12+25)-(11+17) =+37-28 =\blue 9$
      \item $B = (-2,1) + (-9) + (+6,4) + (-8,3)$ \\
         \quad $=+(6,4)-(2,1+9+8,3) =+6,4-19,4 =\blue -13$
      \item $C = (+14) + (-7) + (+2) + (-3,75) + (-5,25)$ \\
         \quad $=+(14+2)-(7+3,75+5,25) =+16-16 ={\blue 0}$
      \item $D = (+13,5) + (-8,1) + (-6,9) + (-5,5)$ \\
         $=+(13,5)-(8,1+6,9+5,5) =+13,5-20,5 =\blue -7$
      \item $E=(-7)+(+1)+(-10) =+(1)-(7+10)$ \\
        \qquad\; $=+1-17 =\blue -16$
   \end{itemize}
\end{corrige}
