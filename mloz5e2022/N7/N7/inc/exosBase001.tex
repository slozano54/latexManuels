\begin{exercice*}
   Effectuer les calculs suivants :   
      \begin{itemize}
         \item[] $A =(-12)+(-15)$
         \item[] $B =(-20)+(+18)$
         \item[] $C =(+21)+(-21)$
         \item[] $D =(+10)+(-13)$
         \item[] $E =(-3)+(+16)$
         \item[] $F =(+13)+(+7)$
         \item[] $G =(+2,1)+(+0,8)$
         \item[] $H =(-1,5)+(-0,1)$
      \end{itemize}   
   \hrefMathalea{https://coopmaths.fr/alea/?uuid=cbc26&id=5R20&n=5&d=10&s=20&s2=false&s3=false&alea=U04q&cd=1&v=eleve&es=02110}
\end{exercice*}

\begin{corrige}
   \ \\ [-5mm]
   \begin{itemize}
      \def\item{}
      \item $A =(-12)+(-15) =-(12+15) ={\red -27}$ \smallskip
      \item $B =(-20)+(+18) =-(20-18) ={\red -2}$ \smallskip
      \item $C =(+21)+(-21) =21-21 ={\red 0}$ \smallskip
      \item $D =(+10)+(-13) =-(13-10) ={\red -3}$ \smallskip
      \item $E =(-3)+(+16) =+(16-3) ={\red 13}$ \smallskip
      \item $F =(+13)+(+7) =+(13+7) ={\red 20}$ \smallskip
      \item $G =(+2,1)+(+0,8) =+(2,1+0,8) ={\red 2,9}$ \smallskip
      \item $H =(-1,5)+(-0,1) =-(1,5+0,1) ={\red -1,6}$
   \end{itemize}
\end{corrige}
