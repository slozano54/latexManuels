\begin{exercice*}
   Écrire chaque fraction sous la forme de la somme d'un nombre entier et d'une fraction inférieure à 1 en posant la division euclidienne. En déduire une écriture comme la soustraction d'un entier et d'une fraction.   
      \begin{multicols}{3}
         \begin{enumerate}
            \begin{spacing}{2}
               \item $\dfrac{27}2$
               \item $\dfrac{92}8$
               \item $\dfrac{13}{3}$
               \item $\dfrac{53}{3}$
               \item $\dfrac{121}{5}$
               \item $\dfrac{127}{4}$
            \end{spacing}
         \end{enumerate}
      \end{multicols}   
\end{exercice*}

\begin{corrige}
   \ \\ [-5mm]
   \begin{enumerate}
      \item \opidiv[voperation=top]{27}{2} donc, $27 =13\times2+1$ \\ [1mm]
         $\dfrac{27}2 =\red 13+\dfrac12 =14-\dfrac12$ \medskip
      \item \opidiv[voperation=top]{92}{8} donc, $92 =11\times8+4$ \\ [1mm]
         $\dfrac{92}8 =\red 11+\dfrac48 =12-\dfrac48$ \medskip
      \item \opidiv[voperation=top]{13}{3} donc, $13 =4\times3+1$ \\ [1mm]
         $\dfrac{13}{3} =\red 4+\dfrac13 =5-\dfrac23$ \medskip
      \item \opidiv[voperation=top]{53}{3} donc, $53 =17\times3+2$ \\ [1mm]
         $\dfrac{53}{3} =\red 17+\dfrac23 =18-\dfrac13$ \medskip
      \item \opidiv[voperation=top]{121}{5} donc, $121 =24\times5+1$ \\ [1mm]
         $\dfrac{121}{5} =\red 24+\dfrac15 =25-\dfrac45$ \medskip
      \item \opidiv[voperation=top]{127}{4} donc, $127 =31\times4+3$ \\ [1mm]
         $\dfrac{127}{4} =\red 31+\dfrac34 =32-\dfrac14$
   \end{enumerate}
\end{corrige}
