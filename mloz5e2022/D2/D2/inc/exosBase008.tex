\begin{exercice*}
    \begin{enumerate}
        \item Lors d'un match de hand-ball, l'équipe qui reçoit un adversaire a une probabilité de $\num{0.21}$ de gagner son match et $\num{0.6}$ de faire un match nul.
              \begin{enumerate}
                  \item Quelle est la probabilité, pour cette équipe, de ne pas perdre le match ?
                  \item Quelle est la probabilité, pour cette équipe, de perdre le match
              \end{enumerate}
        \item Lors d'un match de football, l'équipe qui reçoit un adversaire a une probabilité de $\num{0.4}$ de gagner son match et $\num{0.24}$ de faire un match nul.
              \begin{enumerate}
                  \item Quelle est la probabilité, pour cette équipe, de ne pas perdre le match ?
                  \item Quelle est la probabilité, pour cette équipe, de perdre le match ?
              \end{enumerate}
    \end{enumerate}
\end{exercice*}
\begin{corrige}
    \begin{enumerate}
        \item
              \begin{enumerate}
                  \item Pour ne pas perdre un match, il faut, soit le gagner, soit faire un match nul.
                        \[\num{0.21}+\num{0.6}=\num{0.81}\]
                        La probabilité de \og{}Ne pas perdre le match\fg{} vaut donc \textcolor{red}{\num{0.81}}.

                  \item L'évènement \og{} Perdre le match \fg{} est l'évènement contraire de  \og{} Ne pas perdre le match \fg{}. 
                  
                  La somme de leurs probabilités vaut donc 1.
                        \[1-\num{0.81}=\num{0.19}\]
                        La probabilité de \og{}Perdre le match\fg{} vaut donc \textcolor{red}{\num{0.19}}.
              \end{enumerate}
        \setcounter{enumi}{1}
        \item
              \begin{enumerate}
                  \item Pour ne pas perdre un match, il faut, soit le gagner, soit faire un match nul.
                        \[\num{0.4}+\num{0.24}=\num{0.64}\]
                        La probabilité de \og{}Ne pas perdre le match\fg{} vaut donc \textcolor{red}{\num{0.64}}.

                  \item L'évènement \og{} Perdre le match \fg{} est l'évènement contraire de  \og{} Ne pas perdre le match \fg{}. 
                  
                  La somme de leurs probabilités vaut donc 1.
                        \[1-\num{0.64}=\num{0.36}\]
                        La probabilité de \og{}Perdre le match\fg{} vaut donc \textcolor{red}{\num{0.36}}.
              \end{enumerate}
    \end{enumerate}
\end{corrige}
