\begin{exercice}
    Donner tous les entiers relatifs compris entre :
    \begin{enumerate}
       \item $-2,3$ et $+5,7$.
       \item $-20$ et $-14,8$.
    \end{enumerate}
 \end{exercice}
 
 \begin{corrige}
    \ \\ [-5mm]
    \begin{enumerate}
       \item Entre $-2$ et $+5$ : {\red$-1 \; ; \; 0 \; ; \; 1 \; ; \; 2 \; ; \; 3 \; ; \; 4$}. \smallskip
       \item Entre $-15$ et $-20$ : {\red $-19 \; ; \; -18 \; ; \; -17 \; ; \; -16$}. \smallskip
    \end{enumerate}
 \end{corrige}
